PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX stock: <https://dcm.ffclrp.usp.br/lssb/stock-market-ontology#>

SELECT DISTINCT ?ticker
WHERE {
  ?empresa stock:atuaEm #SETOR_URI# .
  ?empresa stock:temValorMobiliarioNegociado ?acao .
  ?acao stock:temCodigoNegociacao ?ticker .
}
ORDER BY ?ticker
